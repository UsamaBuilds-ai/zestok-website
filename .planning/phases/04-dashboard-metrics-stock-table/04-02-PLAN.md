2. **Export `renderStockTable` function** (no arguments — uses module-level `_balances` and `_searchTerm`):
   ```javascript
   export function renderStockTable() {
     const filtered = _searchTerm
       ? _balances.filter((item) =>
           keyFor(`${item.item} ${item.category}`).includes(keyFor(_searchTerm))
         )
       : _balances;

     const tbody = document.getElementById('stock-table-body');

     if (filtered.length === 0) {
       tbody.innerHTML =
         '<tr><td class="empty-row" colspan="7">No stock balance found</td></tr>';
       return;
     }

     tbody.innerHTML = filtered
       .map(
         (item) => `
           <tr>
             <td><strong>${escapeHtml(item.item)}</strong></td>
             <td>${escapeHtml(item.category)}</td>
             <td>${formatQty(item.inQty)}</td>
             <td>${formatQty(item.outQty)}</td>
             <td><strong>${formatQty(item.balance)}</strong></td>
             <td>${formatRate(item.latestRate)}</td>
             <td>${formatRate(item.value)}</td>
           </tr>`
       )
       .join('');
   }
   ```

3. **Add `debounce` utility function** (module scope):
   ```javascript
   function debounce(fn, ms = 300) {
     let timer;
     return (...args) => {
       clearTimeout(timer);
       if (_debounceTimer) clearTimeout(_debounceTimer);
       timer = setTimeout(() => fn(...args), ms);
       _debounceTimer = timer;
     };
   }
   ```

4. **Create and export `initSearch()` function** — wires the search input to the debounced filter:
   ```javascript
   export function initSearch() {
     const input = document.getElementById('dashboard-search');
     if (!input) return;

     const debouncedFilter = debounce((searchTerm) => {
       _searchTerm = searchTerm;
       renderStockTable();
     }, 300);

     input.addEventListener('input', (e) => {
       debouncedFilter(e.target.value);
       // If input is empty, restore immediately (no debounce wait, per D-48)
       if (!e.target.value) {
         _searchTerm = '';
         clearTimeout(_debounceTimer);
         renderStockTable();
       }
     });

     // Dismiss keyboard on tap outside (D-49)
     document.addEventListener('touchstart', (e) => {
       if (e.target !== input) {
         input.blur();
       }
     });
   }
   ```

5. **Update `showDashboard()`** to call `initSearch()` on first call:
   ```javascript
   let _searchInitialized = false;

   export async function showDashboard() {
     document.getElementById('dashboard-view').classList.remove('hidden');
     if (!_searchInitialized) {
       initSearch();
       _searchInitialized = true;
     }
     await loadDashboard();
   }
   ```

6. **Update `hideDashboard()`** to clear search state:

   ```javascript
   export function hideDashboard() {
     document.getElementById('dashboard-view').classList.add('hidden');
     if (_debounceTimer) {
       clearTimeout(_debounceTimer);
       _debounceTimer = null;
     }
     _searchTerm = '';
   }
   ```

7. **D-49: Keyboard dismiss on clear** — already handled in `initSearch()`: when the input is cleared, `input.blur()` is called after a brief delay to let the UI update first. Add:
