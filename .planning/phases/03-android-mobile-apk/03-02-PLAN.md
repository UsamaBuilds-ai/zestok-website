    **Step 3 — Data fetching function:**
    ```typescript
    const fetchData = async (isRefresh = false) => {
      if (isRefresh) {
        setRefreshing(true);
      } else {
        setLoading(true);
      }
      setError(null);
      
      try {
        const data = await fetchStock(serverIP, pin);
        setStockData(data);
        setLastUpdated(new Date());
      } catch (err: any) {
        if (err.type === 'timeout') {
          setError('Connection timed out. Server may be unreachable.');
        } else if (err.type === 'network') {
          setError('Cannot connect to server. Check your network connection.');
        } else if (err.type === 'http' && err.statusCode === 401) {
          setError('Session expired. Please re-enter your PIN.');
          // Signal parent to re-authenticate
          onSessionExpired?.();
        } else {
          setError('An unexpected error occurred.');
        }
      } finally {
        setLoading(false);
        setRefreshing(false);
      }
    };
    ```

    **Step 4 — Search filtering:**

    Use `useMemo` for efficient filtering (cached until stockData or searchText changes):
    ```typescript
    const filteredData = useMemo(() => {
      if (!stockData) return {};
      if (!searchText.trim()) return stockData;
      
      const lower = searchText.toLowerCase().trim();
      const result: StockResponse = {};
      
      for (const [category, items] of Object.entries(stockData)) {
        const filtered = items.filter(item => 
          item.item.toLowerCase().includes(lower)
        );
        if (filtered.length > 0) {
          result[category] = filtered;
        } else {
          // D-24: Empty category → show header with no items
          result[category] = [];
        }
      }
      return result;
    }, [stockData, searchText]);
    ```
    **Why per-category filtering instead of removing empty categories (D-24):** 
    The user explicitly decided D-24: "Empty stock — Show category headers but no items under them." So every category from the API response is preserved even when empty.

    **Step 5 — Initial data fetch on mount:**

    **Step 8 — Helper: getTimeAgo function (RESEARCH.md Code Examples):**
    ```typescript
    function getTimeAgo(date: Date): string {
      const seconds = Math.floor((Date.now() - date.getTime()) / 1000);
      if (seconds < 60) return 'just now';
      const minutes = Math.floor(seconds / 60);
      if (minutes === 1) return '1 min ago';
      if (minutes < 60) return `${minutes} min ago`;
      return `${Math.floor(minutes / 60)}h ${minutes % 60}m ago`;
    }
    ```

    **Step 9 — Auto-refresh timer component (D-18 compliance):**

    D-18 says "No timed auto-refresh while app is open — on-demand only." But the "last updated" timestamp needs to update text every ~30s to show accurate relative time. This is NOT a data refresh — it's purely a UI update:

    ```typescript
    function LastUpdatedTimer({ lastUpdated }: { lastUpdated: Date }) {
      const [, setTick] = useState(0);
      useEffect(() => {
        const interval = setInterval(() => setTick(t => t + 1), 30000);
        return () => clearInterval(interval);
      }, []);
      return null; // This component only triggers re-render for getTimeAgo update
    }
    ```

    - Do NOT import from `react-router-dom` — StockList is rendered via state machine in App.tsx, not a route
    - Do NOT set `animated={true}` on IonSearchbar — false per Research Code Examples (prevents animation glitches on mobile)

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-07 | Information Disclosure | StockList (search) | mitigate | Search is client-side only (useMemo filter). Search text input never sent to server — no search term leakage |
