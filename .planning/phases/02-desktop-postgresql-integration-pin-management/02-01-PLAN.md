    Port conflict handling function (D-11):
    ```javascript
    async function startExpressWithRetry() {
      const PORT = parseInt(process.env.PORT, 10) || 3000;
      const { startServer } = require('./server');
      
      for (let attempt = 0; attempt < 3; attempt++) {
        try {
          const actualPort = await startServer(PORT);
          const mainWindow = BrowserWindow.getAllWindows()[0];
          if (mainWindow) mainWindow.webContents.send('server:status-update', { running: true, port: actualPort });
          return;
        } catch (err) {
          if (err.code === 'EADDRINUSE') {
            const result = await dialog.showMessageBox({
              type: 'warning',
              title: 'Port In Use',
              message: `Port ${PORT} is already in use.`,
              detail: 'The Express server cannot start on this port. What would you like to do?',
              buttons: ['Retry', 'Use Alternative Port', 'Cancel'],
              defaultId: 0,
              cancelId: 2,
            });
            if (result.response === 1) {
              const altPort = PORT + 1;
              const actualPort = await startServer(altPort);
              // Send status to renderer
              return;
            } else if (result.response === 2) {
              return; // Cancel — run without Express (D-11)
            }
            await new Promise(r => setTimeout(r, 1000));
          } else {
            throw err;
          }
        }
      }
    }
    ```

    Lifecycle cleanup (D-10, D-15):
    ```javascript
    app.on('will-quit', async (event) => {
      event.preventDefault();
      try {
        await require('./server').stopServer();   // 1. Stop Express (D-10)
        const { pool } = require('./db/pool');
        await autoExportToJson();                 // 2. Auto-export to JSON (D-15)
        await pool.end();                         // 3. Close pool
      } catch (err) {
        console.error('Cleanup error:', err.message);
      }
    });
    
    // Windows shutdown handling (Pitfall 2: "before-quit not firing on Windows shutdown")
    app.on('session-end', async () => {
      // On Windows session end, will-quit may not fire reliably
      await require('./server').stopServer();
      const { pool } = require('./db/pool');
      await autoExportToJson();
      await pool.end();
    });
    ```

    Auto-export function (D-15):
    ```javascript
    async function autoExportToJson() {
      try {
        const { pool } = require('./db/pool');
        const result = await pool.query('SELECT * FROM stock_entries ORDER BY created_at DESC');
        if (store) await store.write({ entries: result.rows });
      } catch (err) {
        console.error('Auto-export failed:', err.message);
      }
    }
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Electron app → PostgreSQL | Database credentials cross localhost TCP/5432 |
| Electron app → local filesystem | .env file with credentials on disk |
| Express server → LAN clients | Mobile devices connect via HTTP on LAN port 3000 |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Information Disclosure | .env file on disk | mitigate | .env added to .gitignore (never committed). Pool reads env vars at runtime — no credential in code |
| T-02-02 | Denial of Service | Express port conflict | mitigate | Proactive port check via net.createServer before app.listen. Retry/Alt/Cancel dialog options per D-11 |
| T-02-03 | Denial of Service | PG pool exhaustion | mitigate | pool.query() auto-releases clients. Transaction uses try/finally/release() pattern. connectionTimeoutMillis=2000 for fail-fast |
| T-02-04 | Tampering | SQL injection in server.js | mitigate | /api/pin/status uses parameterized query ($1) — user input never string-interpolated |
| T-02-SC | Tampering | npm package installs | mitigate | pg [SUS] — add blocking human checkpoint before npm install. bcryptjs [OK], dotenv [OK] |
</threat_model>
